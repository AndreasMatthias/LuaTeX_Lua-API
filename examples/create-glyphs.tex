\def\glyph#1{
  \directlua{
    local char = '#1'
    if not char then
      char = 'X'
    end

    local glyph = node.new("glyph")
    glyph.font = font.current()
    glyph.lang = tex.language
    glyph.char = unicode.utf8.byte(char)

    node.write(glyph)
  }
}

Some glyphs:
\glyph{l}
\glyph{o}
\glyph{r}
\glyph{e}
\glyph{m}

German Umlaute:
\glyph{ä}
\glyph{ö}
\glyph{ü}
\end{document}
